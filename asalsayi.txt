#include <stdio.h>

int main() {

    int counter = 0;
        for(int i=5;i<=1000;i++)
        {
            int temp = 0;
            for (int i = 2; i < i; i++)
            {
               if (i % i == 0)
                {
                    temp = 1;
                    break;
                } 
            }
                
            if(temp==0)
            {
                printf("%d\n",i);
                counter++; 
            }
        }
    
    return 0;
}
